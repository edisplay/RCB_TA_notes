Solution to step 2:

```
//self invoking function
//declaring the function and then calling it would be the samething
(function euler010() {
  var maxPrimeValue = 2e+6;
  var i = maxPrimeValue;
  var sum = 0;

  // while primes is less than the sought after number
  while (i >= 2) {
    if (isPrime(i) === true) {
      sum += i;
    }

    i--;
  }

  function isPrime(n) {
    var x = Math.floor(Math.sqrt(n));
    var j = x;

    while (j >= 2) {
      if (n % j === 0) {
        return false;
      }
      j--;
    }

    return true;
  }

  console.log(sum);
  return sum;
}());
```

### 7. Partners do (15 minutes)

sumOddFibs.js
```
var oddSum = 0;
var fibs = [1,2];
var lastFib = 2;
var secondToLastFib = 1;

while (lastFib < 1e6){
  fibs.push(secondToLastFib + lastFib);
  lastFib = fibs[fibs.length-1];
  secondToLastFib = fibs[fibs.length-2];
}

for(var i=0; i < fibs.length; i++){
  if (fibs[i] % 2 != 0){
    oddSum = oddSum + fibs[i];
  }
}

console.log(oddSum);
```

sumOddFibsWithBonus.js
```
var oddSum = 0;
var fibs = [1,2];
var lastFib = 2;
var secondToLastFib = 1;
var type = process.argv[2];

while (lastFib < 1e6){
  fibs.push(secondToLastFib + lastFib);
  lastFib = fibs[fibs.length-1];
  secondToLastFib = fibs[fibs.length-2];
}

for(var i=0; i < fibs.length; i++){
  if (type == 'odd'){
    if (fibs[i] % 2 != 0){
      oddSum = oddSum + fibs[i];
    }
  }else{
    if (fibs[i] % 2 == 0){
      oddSum = oddSum + fibs[i];
    }
  }

}

console.log(oddSum);
```
